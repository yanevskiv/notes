\begin{align*}
    \rho = \frac{\mathrm Q}{\mathrm dV},
    Q' = \frac{Q_L}{L}\\
\end{align*}
\begin{align*}
    \mathrm dQ_L &= \rho\mathrm dV = \rho_0\frac{y}{b} aL\mathrm dy\\
    Q_L &= \int_0^b\rho_0\frac{aL}{b}y\mathrm dy = \frac{\rho_0 aL}{b}\frac{b^2}{2}\\
    Q_L &= \frac{\rho_0 a b L}{2},\,
    Q' = \frac{Q_L}{L} = \frac{\rho_0 ab}{2}
\end{align*}
